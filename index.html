  {
    video: true

    stream = s

Screen sharing

Warning This will only work in Chrome Canary. 


navigator.getUserMedia(
  {
    video: {
      mandatory: {
        chromeMediaSource: "screen"
      }      
    }
  }, 
  function(s){
    stream = s
    video.src = window.URL.createObjectURL(s);
    video.autoplay = true;
  }, 
  // Failed
  function(e){
    console.log(e);
